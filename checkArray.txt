Проверка массива на наличие заданного элемента.
Мое решение
function check(a, x) {
  if (a.includes(x)) {
    return true;  
  } else {
    return false;
  }
}
Другие решения
const check = (a,x) => a.includes(x);

const check = (a, x) => a.includes(x) ? true : false

function check(a, x) {  
  return a.includes(x)
}
